brei

Welcome to Brei's API documentation. There are two ways to use Brei: from the command-line (in which case we refer to the homepage for documentation), or straight from Python. The easiest function to work with is brei(), which links to the command-line app one-to-one.

Program

If you want to read the Program yourself, there are several ways to do so:

  1. Use Program.read(). You give it a Path to a TOML or JSON file and a section, this last bit giving a object path into the data. For instance: Program.read(Path("pyproject.toml"), "tool.brei").
  2. Read your own data format into JSON compatible data, then construct(Program, data). The construct function uses the type annotations in dataclasses to validate the input data.

After reading the data, you'll want to resolve all tasks, i.e. perform includes and run any necessary task to resolve the targets of all other tasks.

program = Program.read(Path("brei.toml"))
db: TaskDB = await resolve_tasks(program)
await db.run(Phony("all"))

There are three kinds of targets: pathlib.Path, Phony and Variable.

121    async def run(self, t: T, visited: dict[T, None] | None = None, **kwargs) -> Result[R]:
122        visited = visited or dict()
123        if t in visited:
124            raise CyclicWorkflowError(list(visited.keys()))
125        visited[t] = None
126
127        if t not in self.index:
128            try:
129                task = self.on_missing(t)
130            except MissingDependency:
131                return MissingFailure(t)
132        else:
133            task = self.index[t]

135        while True:
136            match (result := await task.run_cached(self.run, visited, **kwargs)):
137                case Ok(x) if isinstance(x, Lazy):
138                    task = cast(TaskT, x)
139                case _:
140                    return result

 62    @property
 63    def target_paths(self):

 66    @property
 67    def dependency_paths(self):

 70    def __str__(self):
 71        tgts = ", ".join(str(t) for t in self.creates)
 72        deps = ", ".join(str(t) for t in self.requires)
 73        if self.script is not None:
 74            src = indent(self.script, prefix="  ", predicate=lambda _: True)
 75        elif self.path is not None:
 76            src = str(self.path)
 77        else:
 78            src = " - "
 79        name = f"{self.name}: " if self.name else ""

 82    def __post_init__(self):
 83        if self.name and Phony(self.name) not in self.creates:
 84            self.creates.append(Phony(self.name))
 85        if self.stdin and self.stdin not in self.requires:
 86            self.requires.append(self.stdin)
 87        if self.path and self.path not in self.requires:
 88            self.requires.append(self.path)
 89        if self.stdout and self.stdout not in self.creates:
 90            self.creates.append(self.stdout)

 92    def always_run(self) -> bool:
 93        return len(self.real_requirements) == 0 or len(list(self.target_paths)) == 0

 95    def needs_run(self) -> bool:
 96        if any(not p.exists() for p in self.target_paths):
 97            return True
 98        target_stats = [stat(p) for p in self.target_paths]
 99        dep_stats = [stat(p) for p in self.dependency_paths]
100        if any(t < d for t in target_stats for d in dep_stats):
101            return True

134    async def run(self, *, db: TaskDB):
135        if not self.always_run() and not self.needs_run() and not db.force_run:
136            tgts = " ".join(f"`{t}`" for t in self.target_paths)
137            log.info(f"Targets {tgts} already up-to-date.")
138            return
139
140        log.debug(f"{self}")
141        if (self.path is None and self.script is None):
142            return
143
144        targets = " ".join(f"`{t}`" for t in self.creates)
145        short_note = self.description or (f"#{self.name}" if self.name else None) \
146            or f"creating {targets}"
147        log.info(f"[green]{short_note}[/]", extra={'markup': True})
148
149        stdin: TextIO | int | None = None
150        match self.stdin:
151            case Variable(x):
152                stdin = asyncio.subprocess.PIPE
153                input_data = db.environment[x].encode()
154            case x if isinstance(x, Path):
155                stdin = open(x, "r")
156                input_data = None
157            case _:
158                stdin = None
159                input_data = None

161        if self.runner is None and self.script is not None:
162            if not is_oneliner(self.script):
163                assert self.stdin is None

166                for line in self.script.splitlines():

169                            *shlex.split(line),
170                            stdin=stdin,
171                            stdout=stdout,

173                        )
174                        stdout_data_part, stderr_data = await proc.communicate(input_data)
175                        log.debug(f"return-code {proc.returncode}")
176                    if stdout_data_part:
177                        stdout_data += stdout_data_part
178                    if stderr_data:
179                        log.info(f"[gold1]{short_note}[/] %s", stderr_data.decode().rstrip(), extra={"markup": True})
180
181        elif self.runner is not None:
182            with self.get_script_path() as path, self.get_stdout() as stdout:
183                runner = db.runners[self.runner]
184                args = [string.Template(arg).substitute(script=path) for arg in runner.args]
185                async with db.throttle or nullcontext():
186                    proc = await create_subprocess_exec(
187                        runner.command,
188                        *args,
189                        stdin=stdin,
190                        stdout=stdout,
191                        stderr=asyncio.subprocess.PIPE,
192                    )
193                    stdout_data, stderr_data = await proc.communicate(input_data)
194                    log.debug(f"return-code {proc.returncode}")
195
196            if stderr_data:
197                log.info(f"[gold1]{short_note}[/] %s", stderr_data.decode().rstrip(), extra={"markup": True})
198
199        else:
200            return

146    a program, some tasks may need to be run. Variables that appear in
147    the `creates` field of a task (aka targets), will be resolved eagerly.
148

150    """

155        for var, template in program.environment.items():
156            db.add(TemplateVariable([Variable(var)], [], template))

159        template_index.update(program.template)
160        delayed_calls: list[TemplateCall] = []
161        delayed_templates: list[TaskProxy] = []

166            if c.template not in template_index:
167                log.debug(
168                    "template `%s` not available, waiting for includes to resolve",
169                    c.template,

179            # > if a task has a templated target, those variables should be
180            # > resolvable after all other tasks were added, seeing that the
181            # > task to resolve these variables can't have templated targets
182            # > themselves.
183            if gather_args(tt.all_targets):
